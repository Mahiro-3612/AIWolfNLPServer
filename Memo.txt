AIWolfCommonプロジェクトのorg.aiwolf.common.data.AgentクラスのgetAgentIdx(int)メソッドを下記のように並列処理に対応するように修正しないとバグる
------------------------------------------------------------------------------
	static public Agent getAgent(int idx) {
		synchronized(agentIndexMap){
			if (idx < 0) {
				return null;
			}
			if (!agentIndexMap.containsKey(idx)) {
				Agent agent = new Agent(idx);
				agentIndexMap.put(idx, agent);
			}

			Agent result = agentIndexMap.get(idx);
			//System.err.println("TEST agentIndexMap : " + agentIndexMap);
			return result;
		}
	}
------------------------------------------------------------------------------
